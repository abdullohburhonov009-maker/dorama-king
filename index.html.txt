<script>
    function changeEpisode(episodeNum, fileName) {
        var player = document.getElementById('videoPlayer');
        var source = document.getElementById('videoSource');
        var title = document.getElementById('currentEpisodeName');
        
        title.innerText = "Сейчас играет: Учебная_группа_1_сезон_-_" + episodeNum + "_серия";
        
        source.src = fileName;
        player.load();
        player.play();

        var buttons = document.getElementsByClassName('btn');
        for (var i = 0; i < buttons.length; i++) {
            buttons[i].classList.remove('active');
        }
        buttons[episodeNum - 1].classList.add('active');
    }
</script>
